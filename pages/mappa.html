void setup() {
  lcd.begin(16, 2);                     // Inizializza LCD
  Serial.begin(9600);                   // Avvia comunicazione seriale
  lcd.setCursor(0, 0);                  // Imposta il cursore sulla prima riga
  lcd.print("Tipo rifiuto:");           // Scrive il titolo
  lcd.setCursor(0, 1);                  // Seconda riga
  lcd.print(tipiRifiuto[indiceRifiuto]); // Mostra il primo tipo

  myServo.attach(9, 500, 2500);         // Collega il servo al pin 9
  myServo.write(0);                     // Posizione chiusa

  pinMode(trigPin, OUTPUT);            // Trigger come uscita
  pinMode(echoPin, INPUT);             // Echo come ingresso
  pinMode(buzzerPin, OUTPUT);          // Buzzer in uscita
  pinMode(ledPin, OUTPUT);             // LED in uscita
}



void loop() {



  if (!rifiutoFissato && millis() - tempoUltimoCambio >= intervalloCambio) {
    indiceRifiuto = (indiceRifiuto + 1) % 6;  // Cambia tipo rifiuto
    lcd.setCursor(0, 1);
    lcd.print("                ");           // Cancella riga
    lcd.setCursor(0, 1);
    lcd.print(tipiRifiuto[indiceRifiuto]);   // Mostra nuovo tipo
    tempoUltimoCambio = millis();            // Reset timer
  }



  long durata, distanza;
  digitalWrite(trigPin, LOW);          // Pulizia segnale
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);         // Invia impulso
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  durata = pulseIn(echoPin, HIGH);     // Misura eco
  distanza = durata * 0.034 / 2;       // Calcolo distanza in cm



  if (!rifiutoFissato && distanza > 0 && distanza <= distanzaLimite) {
    rifiutoFissato = true;                   // Blocca tipo scelto
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Rifiuto scelto:");           // Conferma scelta
    lcd.setCursor(0, 1);
    lcd.print(tipiRifiuto[indiceRifiuto]);
  }



  if (distanza > 0 && distanza <= distanzaLimite) {
    myServo.write(90);                      // Apre il coperchio

    if (statoBuzzer == 0) {
      for (int i = 0; i < 5; i++) {
        tone(buzzerPin, 600);               // Suono
        digitalWrite(ledPin, HIGH);         // LED acceso
        delay(100);
        noTone(buzzerPin);                  // Silenzio
        digitalWrite(ledPin, LOW);          // LED spento
        delay(100);
      }
      statoBuzzer = 1;                      // Evita ripetizione
    }

    delay(2000);                            // Attesa
  }



  else {
    myServo.write(0);                       // Chiude coperchio
    statoBuzzer = 0;                        // Resetta stato buzzer
  }
  delay(500);                               // Pausa ciclo
}
