#include <LiquidCrystal.h>
#include <Servo.h>



#define buzzerPin 13
#define rsLCD 12
#define eLCD 11
#define db4LCD 10
#define db5LCD 9
#define db6LCD 8
#define db7LCD 7
#define SERVO 6
#define rLED A0
#define vLED A1
#define bLED A2
#define trigPin 2
#define echoPin 3



LiquidCrystal lcd(rsLCD, eLCD, db4LCD, db5LCD, db6LCD, db7LCD);
Servo myServo;



bool statoBuzzer = 0;
const int distanzaLimite = 25;
String tipiRifiuto[] = {"Carta", "Plastica", "Indiff.", "Umido", "Vetro", "Lattine"};
int indiceRifiuto = 0;
bool rifiutoFissato = false;
unsigned long tempoUltimoCambio = 0;
const unsigned long intervalloCambio = 3000;



void setup() {
  lcd.begin(16, 2);
  Serial.begin(9600);
  lcd.setCursor(0, 0);
  lcd.print("Tipo rifiuto:");
  lcd.setCursor(0, 1);
  lcd.print(tipiRifiuto[indiceRifiuto]);
  myServo.attach(SERVO, 500, 2500);
  myServo.write(0);
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(buzzerPin, OUTPUT);
  pinMode(rLED, OUTPUT);
}



void loop() {



  if (!rifiutoFissato && millis() - tempoUltimoCambio >= intervalloCambio) {
    indiceRifiuto = (indiceRifiuto + 1) % 6;
    lcd.setCursor(0, 1);
    lcd.print("                ");
    lcd.setCursor(0, 1);
    lcd.print(tipiRifiuto[indiceRifiuto]);
    tempoUltimoCambio = millis();
  }



  long durata, distanza;
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  durata = pulseIn(echoPin, HIGH);
  distanza = durata * 0.034 / 2;



  if (!rifiutoFissato && distanza > 0 && distanza <= distanzaLimite) {
    rifiutoFissato = true;
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Rifiuto scelto:");
    lcd.setCursor(0, 1);
    lcd.print(tipiRifiuto[indiceRifiuto]);
  }



  if (distanza > 0 && distanza <= distanzaLimite) {
    myServo.write(90);
    if (statoBuzzer == 0) {
      for (int i = 0; i < 5; i++) {
        tone(buzzerPin, 600);
        analogWrite(rLED, 255);
        delay(100);
        noTone(buzzerPin);
        analogWrite(rLED, 0);
        delay(100);
      }
      statoBuzzer = 1;
    }
    delay(2000);
  } else {
    myServo.write(0);
    statoBuzzer = 0;
  }
  delay(500);
}
