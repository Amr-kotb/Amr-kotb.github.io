#include <LiquidCrystal.h>  // Libreria per il display LCD
#include <Servo.h>          // Libreria per il controllo del servo motore



#define buzzerPin 13            // Pin per il buzzer
#define rsLCD 12                // Pin RS del display LCD
#define eLCD 11                 // Pin E del display LCD
#define db4LCD 10               // Pin DB4 del display LCD
#define db5LCD 9                // Pin DB5 del display LCD
#define db6LCD 8                // Pin DB6 del display LCD
#define db7LCD 7                // Pin DB7 del display LCD
#define SERVO 6                 // Pin del servo motore
#define rLED A0                 // Pin LED rosso
#define vLED A1                 // Pin LED verde
#define bLED A2                 // Pin LED blu
#define trigPin 2              // Pin del sensore ultrasonico (Trigger)
#define echoPin 3              // Pin del sensore ultrasonico (Echo)



LiquidCrystal lcd(rsLCD, eLCD, db4LCD, db5LCD, db6LCD, db7LCD);  // Oggetto per il display LCD
Servo myServo;  // Oggetto per il servo motore



bool statoBuzzer = 0;                   // Stato del buzzer (acceso/spento)
const int distanzaLimite = 25;             // Limite di distanza per rilevare un rifiuto
String tipiRifiuto[] = {"Carta", "Plastica", "Indiff.", "Umido", "Vetro", "Lattine"}; // Array di tipi di rifiuti
int indiceRifiuto = 0;                            // Indice del tipo di rifiuto selezionato
bool rifiutoFissato = false;                      // Flag per verificare se il rifiuto √® stato selezionato
unsigned long tempoUltimoCambio = 0;             // Tempo dell'ultimo cambio di tipo rifiuto
const unsigned long intervalloCambio = 3000;     // Intervallo di tempo per cambiare tipo rifiuto



void setup() {
  lcd.begin(16, 2);                  // Inizializza il display LCD
  Serial.begin(9600);               // Inizializza la comunicazione seriale
  lcd.setCursor(0, 0);             // Imposta il cursore sulla prima riga
  lcd.print("Tipo rifiuto:");     // Mostra il testo "Tipo rifiuto:"
  lcd.setCursor(0, 1);                    // Imposta il cursore sulla seconda riga
  lcd.print(tipiRifiuto[indiceRifiuto]);     // Mostra il primo tipo di rifiuto
  myServo.attach(SERVO, 500, 2500);          // Collega il servo al pin specificato
  myServo.write(0);                      // Posiziona il servo a 0 gradi
  pinMode(trigPin, OUTPUT);                // Imposta il pin del trigger come uscita
  pinMode(echoPin, INPUT);           // Imposta il pin dell'echo come ingresso
  pinMode(buzzerPin, OUTPUT);         // Imposta il pin del buzzer come uscita
  pinMode(rLED, OUTPUT);           // Imposta il pin del LED rosso come uscita
}



void loop() {



  // Cambia tipo di rifiuto ogni intervallo di tempo
  if (!rifiutoFissato && millis() - tempoUltimoCambio >= intervalloCambio) {
    indiceRifiuto = (indiceRifiuto + 1) % 6;                  // Seleziona il prossimo tipo di rifiuto
    lcd.setCursor(0, 1);                                      // Posiziona il cursore sulla seconda riga
    lcd.print("                ");                            // Cancella la riga
    lcd.setCursor(0, 1);                                      // Ritorna al primo carattere della seconda riga
    lcd.print(tipiRifiuto[indiceRifiuto]);                     // Mostra il nuovo tipo di rifiuto
    tempoUltimoCambio = millis();                             // Aggiorna il tempo dell'ultimo cambio
  } 



  long durata, distanza;
  // Misura la distanza utilizzando il sensore ultrasonico
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  durata = pulseIn(echoPin, HIGH);                  // Calcola la durata dell'eco
  distanza = durata * 0.034 / 2;                  // Calcola la distanza in cm



  // Se la distanza √® valida, seleziona il rifiuto
  if (!rifiutoFissato && distanza > 0 && distanza <= distanzaLimite) {
    rifiutoFissato = true;                // Imposta il rifiuto come selezionato
    lcd.clear();                         // Cancella il display LCD
    lcd.setCursor(0, 0);                 // Posiziona il cursore sulla prima riga
    lcd.print("Rifiuto scelto:");          // Mostra il testo "Rifiuto scelto:"
    lcd.setCursor(0, 1);                     // Posiziona il cursore sulla seconda riga
    lcd.print(tipiRifiuto[indiceRifiuto]);        // Mostra il tipo di rifiuto selezionato
  }



  // Azione del servo e del buzzer quando un rifiuto √® selezionato
  if (distanza > 0 && distanza <= distanzaLimite) {
    myServo.write(90);               // Muove il servo a 90 gradi
    if (statoBuzzer == 0) {
      for (int i = 0; i < 5; i++) {
        tone(buzzerPin, 600);             // Emessa un suono dal buzzer
        analogWrite(rLED, 255);            // Accende il LED rosso
        delay(100);
        noTone(buzzerPin);                // Spegne il buzzer
        analogWrite(rLED, 0);                // Spegne il LED rosso
        delay(100);
      }
      statoBuzzer = 1;               // Imposta lo stato del buzzer su acceso
    }
    delay(2000);                // Attende 2 secondi
  } else {
    myServo.write(0);          // Riporta il servo a 0 gradi
    statoBuzzer = 0;               // Imposta lo stato del buzzer su spento
  }
  delay(500);           // Attende mezzo secondo
}
